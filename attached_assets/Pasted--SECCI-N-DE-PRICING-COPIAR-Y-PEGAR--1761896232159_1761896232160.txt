<script>
    function togglePricingPlan(type) {
        const buttons = document.querySelectorAll('.pricing-toggle-btn');
        buttons.forEach(btn => btn.classList.remove('active'));
        event.target.classList.add('active');

        const prices = document.querySelectorAll('.pricing-annual-price');
        const monthlyPrices = [10, 20, 30];
        const annualPrices = [8, 15, 25];

        prices.forEach((price, index) => {
            if (type === 'monthly') {
                price.textContent = monthlyPrices[index];
            } else {
                price.textContent = annualPrices[index];
            }
        });

        const savingsBadges = document.querySelectorAll('.pricing-annual-only');
        savingsBadges.forEach(badge => {
            badge.style.display = type === 'annual' ? 'inline-block' : 'none';
        });
    }
</script>
